```python
import glob

path = r'/kaggle/input/kaggle-survey-2020' 
all_files = glob.glob(path + r"/*.csv")

total = pd.DataFrame()


for file_name in all_files:

  temp = pd.read_csv(file_name, sep=',', dtype='str')

  total = pd.concat([total, temp], axis=0)

print(total.info())
```
